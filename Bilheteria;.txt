CREATE DATABASE Bilheteria;
 USE Bilheteria;

 DROP TABLE IF EXISTS Bilhete;
 DROP TABLE IF EXISTS Tipo_Bilhete;
 DROP TABLE IF EXISTS Compra;
 DROP TABLE IF EXISTS Cliente;

 CREATE TABLE Cliente (
    CPF INT NOT NULL,
    Nome VARCHAR(60) NOT NULL,
    Data_nasc DATE,
    Tel VARCHAR(20),
    Email VARCHAR(80) NOT NULL,
    CONSTRAINT PK_Cliente PRIMARY KEY (CPF)
 );

 CREATE TABLE Compra (
    ID_compra VARCHAR(5) NOT NULL,
    Valor_compra DECIMAL(10,2) NOT NULL,
    Desconto DECIMAL(10,2),
    Credito_prepago DECIMAL(10,2),
    Status VARCHAR(30),
    Pacote_promocional VARCHAR(60),
    CPF INT NOT NULL,
    CONSTRAINT PK_Compra PRIMARY KEY (ID_compra),
    CONSTRAINT FK_Compra_Cliente FOREIGN KEY (CPF)
        REFERENCES Cliente (CPF)
 );

 CREATE TABLE Tipo_Bilhete (
    ID_tipo_bilhete INT NOT NULL,
    Nome_tipo VARCHAR(40) NOT NULL,
    CONSTRAINT PK_TipoBilhete PRIMARY KEY (ID_tipo_bilhete)
 );

 CREATE TABLE Bilhete (
    ID_bilhete VARCHAR(5) NOT NULL,
    Valor_bilhete DECIMAL(10,2) NOT NULL,
    Credito DECIMAL(10,2),
    ID_compra VARCHAR(5) NOT NULL,
    ID_tipo_bilhete INT NOT NULL,
    CONSTRAINT PK_Bilhete PRIMARY KEY (ID_bilhete),
    CONSTRAINT FK_Bilhete_Compra FOREIGN KEY (ID_compra)
        REFERENCES Compra (ID_compra),
    CONSTRAINT FK_Bilhete_Tipo FOREIGN KEY (ID_tipo_bilhete)
        REFERENCES Tipo_Bilhete (ID_tipo_bilhete)
 );

CREATE TRIGGER trg_SomaValorCompra
 ON Bilhete AFTER INSERT, UPDATE
 AS BEGIN UPDATE c
    SET c.Valor_compra =
        (SELECT SUM(b.Valor_bilhete)
         FROM Bilhete b
         WHERE b.ID_compra = c.ID_compra)
    FROM Compra c INNER JOIN inserted i
        ON c.ID_compra = i.ID_compra;
 END;
 GO

 CREATE VIEW vw_ResumoVendas
 AS SELECT
    c.Nome AS Cliente,
    cp.ID_compra,
    tb.Nome_tipo AS Tipo_Bilhete,
    SUM(b.Valor_bilhete) AS Soma_Bilhetes,
    cp.Desconto,
    cp.Valor_compra AS Valor_Final
 FROM Cliente c
 INNER JOIN Compra cp ON c.CPF = cp.CPF
 INNER JOIN Bilhete b ON cp.ID_compra = b.ID_compra
 INNER JOIN Tipo_Bilhete tb ON tb.ID_tipo_bilhete = b.ID_tipo_bilhete
 GROUP BY
    c.Nome, cp.ID_compra, tb.Nome_tipo,
    cp.Valor_compra, cp.Desconto;
