CREATE TABLE Cliente(
    CPF INT NOT NULL PRIMARY KEY,
    Nome VARCHAR(60) NOT NULL,
    Data_Nascimento DATE NULL,
    Telefone VARCHAR(20) NULL,
    Email VARCHAR(80) NULL
);
GO

CREATE TABLE Pagamento(
    id_pagamento INT IDENTITY(1,1) PRIMARY KEY,
    valor DECIMAL(10,2) NOT NULL DEFAULT 0
);
GO

CREATE TABLE BilheteQuartoDoPacote(
    id_reserva INT NOT NULL PRIMARY KEY,
    tipo_quarto VARCHAR(50) NOT NULL,
    data_entrada DATETIME NULL,
    data_saida DATETIME NULL,
    CPF INT NOT NULL,
    id_pagamento INT NOT NULL,
    valorRes DECIMAL(10,2) NOT NULL DEFAULT 0,
    CONSTRAINT FK_Pacote_Cliente FOREIGN KEY (CPF) REFERENCES Cliente(CPF),
    CONSTRAINT FK_Pacote_Pag FOREIGN KEY (id_pagamento) REFERENCES Pagamento(id_pagamento)
);
GO

CREATE TABLE quantiaTotal(
    idPacote INT NOT NULL PRIMARY KEY,
    CPF INT NOT NULL,
    id_pagamento INT NOT NULL,
    custoHotel DECIMAL(10,2) NOT NULL,
    custoBilhete DECIMAL(10,2) NOT NULL,
    custoPacote AS (custoHotel + custoBilhete),
    CONSTRAINT FK_QT_Cliente FOREIGN KEY (CPF) REFERENCES Cliente(CPF),
    CONSTRAINT FK_QT_Pag FOREIGN KEY (id_pagamento) REFERENCES Pagamento(id_pagamento)
);
GO

/* ============================
   Trigger: sincroniza Cliente
   - Copia Cliente para Hotel.dbo.Cliente e Bilheteria.dbo.Cliente se não existirem
   - Set-based
   ============================ */
IF OBJECT_ID('dbo.TG_sync_cliente_pacote') IS NOT NULL
    DROP TRIGGER dbo.TG_sync_cliente_pacote;
GO

CREATE TRIGGER dbo.TG_sync_cliente_pacote
ON dbo.Cliente
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;

    -- Copia para Hotel.dbo.Cliente se o DB/tabela existir e cliente ainda não existir lá
    IF EXISTS (SELECT 1 FROM sys.databases WHERE name = 'Hotel')
      AND EXISTS (SELECT 1 FROM Hotel.sys.tables WHERE name = 'Cliente')
    BEGIN
        INSERT INTO Hotel.dbo.Cliente (CPF, Nome, Data_Nascimento, Telefone, Email)
        SELECT i.CPF, i.Nome, i.Data_Nascimento, i.Telefone, i.Email
        FROM inserted i
        WHERE NOT EXISTS (
            SELECT 1 FROM Hotel.dbo.Cliente h WHERE h.CPF = i.CPF
        );
    END

    -- Copia para Bilheteria.dbo.Cliente se o DB/tabela existir e cliente ainda não existir lá
    IF EXISTS (SELECT 1 FROM sys.databases WHERE name = 'Bilheteria')
      AND EXISTS (SELECT 1 FROM Bilheteria.sys.tables WHERE name = 'Cliente')
    BEGIN
        INSERT INTO Bilheteria.dbo.Cliente (CPF, Nome, Data_nasc, Tel, Email)
        SELECT i.CPF, i.Nome, i.Data_Nascimento, i.Telefone, i.Email
        FROM inserted i
        WHERE NOT EXISTS (
            SELECT 1 FROM Bilheteria.dbo.Cliente b WHERE b.CPF = i.CPF
        );
    END
END;
GO

/* =============================================================================
   Trigger: ao inserir BilheteQuartoDoPacote
   - Cria pagamentos e reservas no Hotel (gera id_pagamento em Hotel)
   - Cria Compra + Bilhete na Bilheteria (gera ID_compra e ID_bilhete como VARCHAR(5))
   - Implementado de forma set-based; gera IDs com MAX + ROW_NUMBER e bloqueio de tabela
   ============================================================================= */
IF OBJECT_ID('dbo.TG_sync_reserva_pacote') IS NOT NULL
    DROP TRIGGER dbo.TG_sync_reserva_pacote;
GO

CREATE TRIGGER dbo.TG_sync_reserva_pacote
ON dbo.BilheteQuartoDoPacote
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT (
        EXISTS (SELECT 1 FROM sys.databases WHERE name='Hotel') AND
        EXISTS (SELECT 1 FROM sys.databases WHERE name='Bilheteria')
    )
    BEGIN
        RETURN;
    END

    BEGIN TRY
        BEGIN TRAN;

        SELECT 
            i.*,
            ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS rn
        INTO #ins
        FROM inserted i;

        DECLARE @maxHotelPag INT;
        SELECT @maxHotelPag = ISNULL(MAX(id_pagamento), 0)
        FROM Hotel.dbo.Pagamento WITH (TABLOCKX, HOLDLOCK);

        INSERT INTO Hotel.dbo.Pagamento (id_pagamento, valor)
        SELECT @maxHotelPag + rn, valorRes
        FROM #ins;

        INSERT INTO Hotel.dbo.Reserva
            (id_reserva, tipo_quarto, data_entrada, data_saida, CPF, valorRes, id_pagamento)
        SELECT 
            id_reserva,
            tipo_quarto,
            data_entrada,
            data_saida,
            CPF,
            valorRes,
            @maxHotelPag + rn
        FROM #ins;

        INSERT INTO Bilheteria.dbo.Cliente (CPF, Nome, Data_nasc, Tel, Email)
        SELECT DISTINCT 
            ins.CPF,
            c.Nome,
            c.Data_Nascimento,
            c.Telefone,
            c.Email
        FROM #ins ins
        LEFT JOIN Bilheteria.dbo.Cliente b ON b.CPF = ins.CPF
        INNER JOIN dbo.Cliente c ON c.CPF = ins.CPF
        WHERE b.CPF IS NULL;

        DECLARE @maxCompra INT;
        SELECT @maxCompra = ISNULL(MAX(CAST(ID_compra AS INT)), 0)
        FROM Bilheteria.dbo.Compra WITH (TABLOCKX, HOLDLOCK);

        SELECT *,
            ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS rn2
        INTO #ins2
        FROM #ins;

        INSERT INTO Bilheteria.dbo.Compra
            (ID_compra, Valor_compra, Desconto, Credito_prepago, Status, Pacote_promocional, CPF)
        SELECT 
            RIGHT('00000' + CAST(@maxCompra + rn2 AS VARCHAR(5)), 5),
            valorRes,
            0,
            0,
            'Aprovado',
            'Pacote Integrado HB',
            CPF
        FROM #ins2;

        INSERT INTO Bilheteria.dbo.Bilhete
            (ID_bilhete, Valor_bilhete, Credito, ID_compra, ID_tipo_bilhete)
        SELECT 
            RIGHT('00000' + CAST(@maxCompra + rn2 AS VARCHAR(5)), 5),
            valorRes,
            0,
            RIGHT('00000' + CAST(@maxCompra + rn2 AS VARCHAR(5)), 5),
            1
        FROM #ins2;

        COMMIT;
    END TRY
    BEGIN CATCH
        IF XACT_STATE() <> 0 ROLLBACK;
        DECLARE @ErrMsg NVARCHAR(4000) = ERROR_MESSAGE();
        RAISERROR('TG_sync_reserva_pacote failed: %s',16,1,@ErrMsg);
    END CATCH
END;
GO


/* ===========================================
   Trigger: quando inserir em quantiaTotal
   -> atualiza Reserva.valorRes somando custoHotel
   =========================================== */
IF OBJECT_ID('dbo.TG_computa_compa_pacote') IS NOT NULL
    DROP TRIGGER dbo.TG_computa_compa_pacote;
GO

CREATE TRIGGER dbo.TG_computa_compa_pacote
ON dbo.quantiaTotal
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE r
    SET r.valorRes = r.valorRes + i.custoHotel
    FROM Hotel.dbo.Reserva r
    INNER JOIN inserted i ON r.CPF = i.CPF;
END;
GO
